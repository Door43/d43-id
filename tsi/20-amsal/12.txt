\c 12
\p 
\p
\b
\q1
\v 1 Orang yang gemar menuntut ilmu dapat menerima teguran,
\q2 tetapi orang yang tidak mau ditegur akan tetap bodoh.
\p 
\b
\q1
\v 2 Orang baik berkenan di hadapan TUHAN,
\q2 tetapi TUHAN mengutuk orang yang merencanakan hal jahat.
\p
\b
\q1
\v 3 Orang tidak dapat mendirikan sesuatu dengan kejahatan,
\q2 tetapi dasar orang benar tidak tergoyahkan.
\p
\b
\q1
\v 4 Isteri yang cakap menjaga kehormatan suaminya,
\q2 tetapi isteri yang membuat malu menghancurkan nama baik suaminya.
\p
\b
\q1
\v 5 Orang benar membuat rencana yang adil,
\q2 tetapi tujuan orang jahat menyesatkan.
\p
\b
\q1
\v 6 Perkataan orang jahat ingin menjebak seolah mengintai darah 
\q2 tetapi perkataan orang benar akan membebaskan mereka dari incaran.
\p
\b
\q1
\v 7 Orang jahat akan digulingkan dan dimusnahkan,
\q2 tetapi keluarga orang benar akan berdiri teguh.
\p
\b
\q1
\v 8 Orang dipuji sepadan dengan akal budinya 
\q2 tetapi orang yang memiliki pikiran yang serong dibenci.
\p
\b
\q1
\v 9 Lebih baik menjadi orang yang diremehkan orang lain tetapi memiliki pekerja
\q2  daripada orang yang berlagak kaya, tetapi tidak mampu untuk membeli makan.
\p
\b
\q1
\v 10 Orang benar memperhatikan hewan peliharaannya, 
\q2 tetapi belas kasihan orang jahat didasari maksud keji.
\p
\b
\q1
\v 11 Siapa yang membanting tulang tidak akan kelaparan, 
\q2 tetapi siapa yang mengejar lamunan tidak berhikmat.
\p
\b
\q1
\v 12 Orang jahat mengingini hasil jarahan orang lain,
\q2 tetapi orang benar akan tetap bertahan dan berkembang.
\p
\b
\q1
\v 13 Orang jahat terjerat oleh perkataannya sendiri, 
\q2 tetapi orang benar terlepas dari kesusahan.
\p
\b
\q1
\v 14 Setiap orang terpuaskan oleh hal baik karena buah perkataannya,   
\q2 dan, setiap orang akan menerima imbalan yang setimpal untuk hasil kerjanya.
\p
\b
\q1
\v 15 Orang bodoh menganggap jalan yang dia tempuh adalah lurus,
\q2 tetapi hanya orang bijak yang mau mendengarkan nasehat.
\p
\b
\q1
\v 16 Orang bodoh segera menyatakan kekesalan dan sakit hati,
\q2 tetapi orang bijak mengabaikan penghinaan.
\p
\b
\q1
\v 17 Siapa yang dapat dipercaya menyatakan yang benar dan yang adil,
\q2 tetapi seorang saksi dusta menyatakan tipu muslihat.
\p
\b
\q1
\v 18 Perkataan yang lancang bagaikan tikaman pedang,
\q2 tetapi lidah orang bijak menyembuhkan.
\p
\b
\q1
\v 19 Bibir yang menyatakan hal yang benar akan tetap untuk selama-lamanya,
\q2 tetapi lidah dusta hanya bertahan sekejap.
\p
\b
\q1
\v 20 Hati orang yang merencanakan jahat dipenuhi tipu daya, 
\q2 tetapi suka cita memenuhi orang yang memberikan bimbingan dan nasehat perdamaian.
\p
\b
\q1
\v 21 Orang benar akan terhindar dari segala bencana,
\q2 tetapi orang fasik mengalami malapetaka.
\p
\b
\q1
\v 22 TUHAN membenci bibir pendusta,
\q2 tetapi yang menyenangkan-Nya ialah siapa yang berlaku jujur.

\p
\b
\q1
\v 23 Orang bijak [menahan diri dari//mendiamkan//berdiam diri akan] pengetahuannya,
\q2 tetapi orang bodoh mengembar-gemborkan kebodohannya.
\p
\b
\q1
\v 24 Tangan orang yang tekun akan memerintah,
\q2 tetapi orang yang malas mendatangkan kerja paksa.
\p
\b
\q1
\v 25 Kekuatiran dalam hati memberatkan seseorang,   
\q2 tetapi perkataan baik membawa suka cita.
\p
\b
\q1
\v 26 Orang benar sangat berhati-hati dalam persahabatan,
\q2 tetapi jalan orang jahat menyesatkan.
\p
\b
\q1
\v 27 Orang malas tidak dapat menangkap hasil buruannya,
\q2 tetapi orang rajin memperoleh harta kekayaan.
\p
\b
\q1
\v 28 Di jalan benar orang menuju kepada hidup,
\q2  tetapi menjejaki jalan selain itu menuju kepada maut.
